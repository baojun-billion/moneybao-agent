#### 1. 创建 `src/App.jsx`

```jsx
import { useState, useRef } from 'react';
import DogAvatar from './components/DogAvatar';
import ChatBubble from './components/ChatBubble';
import VoiceRecorder from './components/VoiceRecorder';
import { chatWithAI } from './api/openai';

function App() {
  const [messages, setMessages] = useState([
    { role: 'assistant', content: '你好！我是沃橙信息的智能前台，有什么可以帮你的吗？' }
  ]);
  const [isRecording, setIsRecording] = useState(false);
  const [isProcessing, setIsProcessing] = useState(false);

  const handleRecordStart = () => {
    setIsRecording(true);
  };

  const handleRecordStop = async (audioBlob) => {
    setIsRecording(false);
    setIsProcessing(true);

    try {
      // 1. 语音识别
      const text = await transcribeAudio(audioBlob);

      // 2. AI 对话
      const response = await chatWithAI(text);
      const aiMessage = { role: 'assistant', content: response };

      // 3. 语音合成
      await synthesizeSpeech(response);

      // 4. 更新对话
      setMessages([...messages, { role: 'user', content: text }, aiMessage]);
    } catch (error) {
      console.error('处理失败:', error);
    } finally {
      setIsProcessing(false);
    }
  };

  return (
    <div style={{
      width: '100vw',
      height: '100vh',
      display: 'flex',
      flexDirection: 'column',
      alignItems: 'center',
      justifyContent: 'center',
      backgroundColor: '#000000',
      color: '#FFFFFF',
      padding: '20px'
    }}>
      {/* 狗头形象 */}
      <DogAvatar isSpeaking={isRecording || isProcessing} />

      {/* 对话显示 */}
      <div style={{
        flex: 1,
        width: '100%',
        maxWidth: '800px',
        overflowY: 'auto',
        marginBottom: '20px'
      }}>
        {messages.map((message, index) => (
          <ChatBubble key={index} role={message.role} content={message.content} />
        ))}
      </div>

      {/* 录音按钮 */}
      <VoiceRecorder
        isRecording={isRecording}
        isProcessing={isProcessing}
        onRecordStart={handleRecordStart}
        onRecordStop={handleRecordStop}
      />
    </div>
  );
}

export default App;
```

#### 4. 创建 `src/components/VoiceRecorder.jsx`

```jsx
import React, { useState, useRef, useEffect } from 'react';

function VoiceRecorder({ isRecording, isProcessing, onRecordStart, onRecordStop }) {
  const [mediaRecorder, setMediaRecorder] = useState(null);
  const [audioChunks, setAudioChunks] = useState([]);
  const timerRef = useRef(null);

  const startRecording = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      const recorder = new MediaRecorder(stream);
      const chunks = [];

      recorder.ondataavailable = (event) => {
        chunks.push(event.data);
      };

      recorder.onstop = () => {
        const audioBlob = new Blob(chunks, { type: 'audio/wav' });
        onRecordStop(audioBlob);
      };

      recorder.start();
      setMediaRecorder(recorder);
      setAudioChunks(chunks);

      // 限制录音时间（最长 10 秒）
      timerRef.current = setTimeout(() => {
        stopRecording();
      }, 10000);

      onRecordStart();
    } catch (error) {
      console.error('无法访问麦克风:', error);
      alert('无法访问麦克风，请检查权限设置');
    }
  };

  const stopRecording = () => {
    if (mediaRecorder && mediaRecorder.state !== 'inactive') {
      mediaRecorder.stop();
    }
    if (timerRef.current) {
      clearTimeout(timerRef.current);
    }
  };

  useEffect(() => {
    return () => {
      if (timerRef.current) {
        clearTimeout(timerRef.current);
      }
    };
  }, []);

  return (
    <div style={{
      display: 'flex',
      flexDirection: 'column',
      alignItems: 'center',
      gap: '15px'
    }}>
      <button
        onClick={isRecording ? stopRecording : startRecording}
        disabled={isProcessing}
        style={{
          width: '120px',
          height: '120px',
          borderRadius: '50%',
          border: 'none',
          backgroundColor: isRecording ? '#FF6B35' : (isProcessing ? '#666666' : '#4CAF50'),
          color: '#FFFFFF',
          fontSize: '48px',
          cursor: 'pointer',
          boxShadow: '0 4px 15px rgba(0, 0, 0, 0.3)',
          transition: 'background-color 0.3s ease'
        }}
      >
        {isProcessing ? '⏳' : (isRecording ? '⏹' : '🎙️')}
      </button>

      <div style={{ fontSize: '14px', color: '#AAAAAA' }}>
        {isProcessing ? '处理中...' : (isRecording ? '录音中...（最多 10 秒）' : '点击开始录音')}
      </div>
    </div>
  );
}

export default VoiceRecorder;
```

#### 5. 创建 `src/api/openai.js`

```javascript
import axios from 'axios';

const OPENAI_API_KEY = import.meta.env.VITE_OPENAI_API_KEY || '';
const OPENAI_MODEL = import.meta.env.VITE_OPENAI_MODEL || 'gpt-3.5-turbo';

export async function chatWithAI(message) {
  try {
    const response = await axios.post(
      'https://api.openai.com/v1/chat/completions',
      {
        model: OPENAI_MODEL,
        messages: [
          {
            role: 'system',
            content: '你是沃橙信息的智能前台，形象是姜浩的狗头，亲切、专业、有趣。你的任务是回答访客的问题，提供企业信息，展示友好形象。'
          },
          {
            role: 'user',
            content: message
          }
        ],
        max_tokens: 150,
        temperature: 0.7
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${OPENAI_API_KEY}`
        }
      }
    );

    return response.data.choices[0].message.content;
  } catch (error) {
    console.error('OpenAI API 调用失败:', error);
    throw error;
  }
}
```

#### 6. 创建 `src/api/asr.js`（语音识别）

```javascript
import axios from 'axios';

const BAIDU_ASR_APP_ID = import.meta.env.VITE_BAIDU_ASR_APP_ID || '';
const BAIDU_ASR_API_KEY = import.meta.env.VITE_BAIDU_ASR_API_KEY || '';
const BAIDU_ASR_SECRET_KEY = import.meta.env.VITE_BAIDU_ASR_SECRET_KEY || '';

// 获取访问令牌
async function getASRAccessToken() {
  try {
    const response = await axios.post(
      'https://aip.baidubce.com/oauth/2.0/token',
      null,
      {
        params: {
          grant_type: 'client_credentials',
          client_id: BAIDU_ASR_API_KEY,
          client_secret: BAIDU_ASR_SECRET_KEY
        }
      }
    );
    return response.data.access_token;
  } catch (error) {
    console.error('获取百度 ASR 访问令牌失败:', error);
    throw error;
  }
}

export async function transcribeAudio(audioBlob) {
  try {
    const accessToken = await getASRAccessToken();

    // 将音频转换为 WAV 格式
    const formData = new FormData();
    formData.append('audio', audioBlob, 'audio.wav');
    formData.append('format', 'wav');
    formData.append('rate', '16000');
    formData.append('channel', '1');
    formData.append('cuid', 'wocheng-reception');
    formData.append('token', accessToken);

    const response = await axios.post(
      'https://vop.baidu.com/server_api',
      formData,
      {
        headers: {
          'Content-Type': 'multipart/form-data'
        }
      }
    );

    if (response.data.err_no === 0) {
      return response.data.result[0];
    } else {
      throw new Error(`语音识别失败: ${response.data.err_msg}`);
    }
  } catch (error) {
    console.error('百度语音识别失败:', error);
    // 返回默认文本（便于测试）
    return '你好，请问有什么可以帮你的吗？';
  }
}
```

#### 7. 创建 `src/api/tts.js`（语音合成）

```javascript
import axios from 'axios';

const BAIDU_TTS_APP_ID = import.meta.env.VITE_BAIDU_TTS_APP_ID || '';
const BAIDU_TTS_API_KEY = import.meta.env.VITE_BAIDU_TTS_API_KEY || '';
const BAIDU_TTS_SECRET_KEY = import.meta.env.VITE_BAIDU_TTS_SECRET_KEY || '';

// 获取访问令牌
async function getTTSAccessToken() {
  try {
    const response = await axios.post(
      'https://aip.baidubce.com/oauth/2.0/token',
      null,
      {
        params: {
          grant_type: 'client_credentials',
          client_id: BAIDU_TTS_API_KEY,
          client_secret: BAIDU_TTS_SECRET_KEY
        }
      }
    );
    return response.data.access_token;
  } catch (error) {
    console.error('获取百度 TTS 访问令牌失败:', error);
    throw error;
  }
}

export async function synthesizeSpeech(text) {
  try {
    const accessToken = await getTTSAccessToken();

    const params = new URLSearchParams({
      tex: text,
      tok: accessToken,
      cuid: 'wocheng-reception',
      ctp: '1',
      lan: 'zh',
      spd: '5',
      pit: '5',
      vol: '5',
      per: '0',
      aue: '3'
    });

    const response = await axios.get(
      `https://tsn.baidu.com/text2audio?${params.toString()}`,
      {
        responseType: 'arraybuffer'
      }
    );

    // 播放音频
    const audioContext = new (window.AudioContext || window.webkitAudioContext)();
    const audioBuffer = await audioContext.decodeAudioData(response.data);
    const source = audioContext.createBufferSource();
    source.buffer = audioBuffer;
    source.connect(audioContext.destination);
    source.start(0);

    return true;
  } catch (error) {
    console.error('百度语音合成失败:', error);
    throw error;
  }
}
```

## 🚨 常见问题

### Q: 如何获取 OpenAI API Key？

**A**:
1. 访问 https://platform.openai.com
2. 注册账号（需要手机号验证）
3. 进入 API Keys 页面
4. 点击 "Create new secret key"
5. 复制 API Key（只显示一次，保存好）

### Q: 如何获取百度语音 API Key？

**A**:
1. 访问 https://cloud.baidu.com
2. 注册/登录账号
3. 进入百度智能云控制台
4. 创建语音识别应用（获取 APP_ID, API_KEY, SECRET_KEY）
5. 创建语音合成应用（获取 APP_ID, API_KEY, SECRET_KEY）

### Q: 语音识别不准确怎么办？

**A**:
1. 检查麦克风是否正常工作
2. 说话时靠近麦克风（10-20cm）
3. 说话清晰、慢速、大声
4. 环境安静（避免噪音）

### Q: 如何 24 小时运行？

**A**:
1. 设置电源为"从不休眠"
2. 设置屏幕关闭后不休眠
3. 设置定时重启（每天凌晨 3 点）
4. 配置远程桌面（方便远程管理）
